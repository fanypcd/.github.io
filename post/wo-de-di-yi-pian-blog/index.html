void qsort(int l, int r)
{
    int i = l, j = r, mid;
    mid = a[(l + r) / 2];
    do
    {
        while(a[i] < mid)
        {
            i++;
        }
        while(a[j] > mid)
        {
            j--;
        }
        if(i <= j)
        {
            swap(a[i], a[j]);
            i++;
            j--;
        }
    }while(i <= j);
    if(l < j)
    {
        qsort(l, j);
    }
    if(i < r)
    {
        qsort(i, r);
    }
    return;
}
